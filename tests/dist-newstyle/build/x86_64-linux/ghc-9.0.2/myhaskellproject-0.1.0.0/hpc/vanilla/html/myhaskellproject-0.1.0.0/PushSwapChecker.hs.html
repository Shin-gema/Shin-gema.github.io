   26 execInstruction :: [Char] -> ([a], [a]) -> ([a], [a])
   27 execInstruction "sa" tulpe = first swap tulpe
   28 execInstruction "sb" tulpe = second swap tulpe

   30 execInstruction "pa" tulpe = pa tulpe

   36 execInstruction "rrb" tulpe = second rl tulpe
