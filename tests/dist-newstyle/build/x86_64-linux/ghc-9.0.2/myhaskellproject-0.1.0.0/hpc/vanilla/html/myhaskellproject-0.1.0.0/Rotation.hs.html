    8 module Rotation where
    9 
   10 rf :: [a] -> [a]
   11 rf [] = []
   12 rf (x:xs) = xs ++ [x]
   13 
   14 rl :: [a] -> [a]
   15 rl [] = []
   16 rl a = 
   17     let lastArg = last a
   18     in lastArg : init a
   19 
   20 rr :: ([a], [a]) -> ([a], [a])
   21 rr ([], []) = ([], [])
   22 rr (a, b) = (rf a, rf b)
   23 
   24 rrr :: ([a], [a]) -> ([a], [a])
   25 rrr ([], []) = ([], [])
   26 rrr (a, b) = (rl a, rl b)
