   10 listComp :: Ord a => [a] -> Bool
   11 listComp [] = True
   12 listComp [_] = True
   13 listComp (a:b:xs)
   14     | a <= b    = listComp (b:xs)
   15     | otherwise = False
   16 
   17 
   18 listChecker :: (Ord a, Show a) => ([a], [a]) -> IO ()
   19 listChecker a
   20     | listComp (fst a) && null (snd a) = putStrLn "OK"
   21     | otherwise = putStrLn ("KO: " ++ show a)
